flight_weather <- dbGetQuery(db, "
SELECT tailnum, dep_delay, w.* FROM flights AS f 
JOIN weather AS w ON w.origin = f.origin AND 
w.month = f.month AND
w.day = f.day AND
w.hour = f.hour") 

# calculate averages for the variables of interest
flight_weather_avg <- flight_weather %>%
  group_by(origin, month, day, hour) %>%
  summarise_at(.vars = c(2, 8:10, 12:16), .funs = mean) %>%
  ungroup()
